const debounce = function(fn, wait, immediate) {
  var timer = null;
  var result = null;
  const debounced = function(...args) {
    if(timer) {
      clearTimeout(timer);
    }
    
    if(immediate) {
      var callNow = !timer;
      timer = setTimeout(() => {
        timer = null;
      }, wait)
      if(callNow) {
        fn.apply(this, args);
      }
    }else {
      timer = setTimeout(() => {
        result = fn.apply(this, args); // 立即执行时， 可能有返回值
      }, wait)
    }
    return result;
  }
  
  return debounced;
}

                                                        /* 双剑合壁版 */
function throttle(func, wait) {
    var timeout, context, args, result;
    var previous = 0;

    var later = function() {
        previous = +new Date();
        timeout = null;
        func.apply(context, args)
    };

    var throttled = function() {
        var now = +new Date();
        //下次触发 func 剩余的时间
        var remaining = wait - (now - previous);
        context = this;
        args = arguments;
         // 如果没有剩余的时间了或者你改了系统时间
        if (remaining <= 0 || remaining > wait) {
            if (timeout) {
                clearTimeout(timeout);
                timeout = null;
            }
            previous = now;
            func.apply(context, args);
        } else if (!timeout) {
            timeout = setTimeout(later, remaining);
        }
    };
    return throttled;
}

                                                        Function.prototype.myApply = function(context) {
  if(typeof this !== 'function') {
        throw new TypeError(`${this} is not funciton`);
    }
  var context = context || window;
  context.fn = this;
  let result;
  if(arguments[1]) { // 此处与call稍有不同,因为apply 至多有两个参数，而call是参数序列
    result = context.fn(...arguments[1]);
  }else {
    result = context.fn();
  }
  delete context.fn;
  return result;
}

                                                        Function.prototype.myBind = function (context) {
    if (typeof this !== 'function') {
        throw new TypeError(`${this} is not function`);
    }
    let that = this;
    let arg = [...arguments].slice(1)
    return function F() {
        // 因为返回了一个函数，我们可以 new F()，所以需要判断
        if (this instanceof F) { // 此时 this 指向实例
            return new that(...arg, ...arguments)
        } else {
            return that.apply(context, [...args, ...arguments])
        }
    }
}

                                                        /**
 * 查找constructor.prototype 是否在 object的原型链上,在就返回true，否则false
 * @param {Object} left 
 * @param {constructor.prototype} right 
 */
const instanceOf = function(left, right) {
    let leftValue = left.__proto__;
    let rightValue = right.prototype;
    while(true) {
        if(leftValue === null) { // leftValue === null || leftValue === undefined
            return false;//原型链的末端
        }
        if(leftValue === rightValue) {
            return true;
        }
        leftValue = leftValue.__proto__;
    }
}

                                                        # 二分查找

                                                        const binarySearch = function(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  while(left <= right) {
    let middle = Math.floor((left + right)/2);
    if(target === arr[middle]) return middle;
    else if(target > arr[middle]) {
      left = middle + 1;
    }else {
      right = middle - 1;
    }
  }
  return -1;
}

                                                        # 快排

                                                        /**
 * 设置一个基准数
 * 递归快排
 * T O(nlogn) ~ O(n^2)
 * S O(0) ~ O(n)
 * @param {Array} arr 
 */
export const quickSort = function(arr) {
    if(arr.length <= 1) {
        return arr;
    }
    let pivot = Math.floor(arr.length/2);
    let midItem = arr.splice(pivot, 1)[0];// 删除从mid位置开始的一个元素，并返回
    let leftArr = [];
    let rightArr = [];
    for(let i=0; i<arr.length; i++) {
        if(arr[i] <= midItem) {
            leftArr.push(arr[i]);
        }else {
            rightArr.push(arr[i]);
        }
    }

    return quickSort(leftArr).concat([midItem], quickSort(rightArr));
}

    for (var i = 0; i < arr.length; i++) {
        for (var j = 0; j < arr.length-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                var temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    };
    return arr;
}

                                                        # 封装ajax

                                                        const ajax = function(options) {
    var options = options || {};
    options.type = (options.type || 'GET').toUpperCase();
    options.dataType = options.dataType || 'json';
    options.async = options.async || true;
    var params = formatParams(options.data);

    var xhr = XMLHttpRequest ? new XMLHttpRequest() : new ActiveXObject('Microsoft.XMLHTTP');
  
    xhr.onreadystatechange = function() {
        if(xhr.readyState === 4) {
          	var res;
            if((xhr.status >= 200 && xhr.status < 300) || xhr.status === 304) {
                if(options.success && typeof options.success === 'function') {
                    res = xhr.responseText;
                    options.success.call(xhr, res);
                }
            }else {
                if (options.fail && typeof options.fail === 'function') {
                    res = xhr.responseText;
                    options.fail.call(xhr, res);
                }
            }
        }
    }

    if(options.type === 'GET') {
        xhr.open(options.type, options.url+'?'+params, options.async);
        xhr.send()
    }else if(options.type === 'POST'){
        xhr.open(options.type, options.url, options.async);
        xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
        xhr.send(params);
    }
}

function formatParams(params) {
    var arr = [];
    for(var param in params) {
        arr.push(encodeURIComponent(param)+ '=' + encodeURIComponent(params[param]));
    }

    return arr.join('&');
}

                                                        const deepClone = function(obj) {
  if(typeof obj !== 'object') {
    return obj;
  }
  
  var newObj = obj instanceof Array ? []:{};
  for(var key in obj) {
     if(Object.prototype.hasOwnProperty.call(obj, key)) {
            newObj[key] = typeof obj[key] === 'object' ? deepClone(obj[key]): obj[key];
     }
  }
  
  return newObj;
}

}

                                                        /* ES5 Array.map实现 */
Array.prototype.mymap = function(fn, context) {
    var arr = Array.prototype.slice.call(this);//由于是ES5所以就不用...展开符了
    var mappedArr = [];
    for (var i = 0; i < arr.length; i++) {
        mappedArr.push(fn.call(context, arr[i], i, this));
    }
    return mappedArr;
}

                                                        # 用setTimeout来实现setInterval

                                                        var timers = [];
function _setInterval(fn, time) {
  var execute = function(fn, time) {
    timers.push(setTimeout(() => {
      fn();
      execute(fn, time);
    }, time))
  }
  execute(fn, time);
}

function _clearInterval(timers) {
  timers.forEach(timer => {
    clearTimeout(timer);
  })
}
